def merge(x,first,mid,last)
  i = first
  q = 0
  j = mid+1
  z = []
  while(i <= mid && j <= last ) do
    if x[i] < x[j]
      z[q] = x[i]
      i += 1
    else
      z[q] = x[j]
      j += 1
    end
    q += 1
  end
  z += x[i..mid] if i <= mid
  z += x[j..last] if j <= last
  x[first..last] = z
end
